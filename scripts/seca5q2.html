import os
import xml.dom.minidom

stream = os.popen("nmap -sS -p10000-20000 -oX - localhost")
output=stream.read()

scan_xml = xml.dom.minidom.parseString(output)
ports = scan_xml.getElementsByTagName("port")
for port in ports:
        poid = port.getAttribute("portid")
	print ("%s/%s %s" % (poid,
                             port.getAttribute("protocol"),
                             port.getElementsByTagName("state")[0].getAttribute("state")))
	cmd = "nc localhost %s < input.txt" % poid
	stream = os.popen(cmd)
        output = stream.read()
	print(output)
